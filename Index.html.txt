    <script type="module">
        import { initializeApp } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-app.js";
        import { getAuth, signInAnonymously, onAuthStateChanged } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-auth.js";
        import { getFirestore, collection, addDoc, getDocs, serverTimestamp } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-firestore.js";

        const firebaseConfig = typeof __firebase_config !== 'undefined' ? JSON.parse(__firebase_config) : {
            apiKey: "",
            authDomain: "",
            projectId: "",
            storageBucket: "",
            messagingSenderId: "",
            appId: ""
        };
        
        const app = initializeApp(firebaseConfig);
        const auth = getAuth(app);
        const db = getFirestore(app);
        const appId = typeof __app_id !== 'undefined' ? __app_id : 'default-app-id';

        lucide.createIcons();

        let user = null;
        signInAnonymously(auth);
        onAuthStateChanged(auth, u => user = u);

        const form = document.getElementById('signup-form');
        const btn = document.getElementById('submit-btn');
        const msg = document.getElementById('form-message');

        form.onsubmit = async (e) => {
            e.preventDefault();
            if (!user) return;
            btn.disabled = true;
            try {
                await addDoc(collection(db, 'artifacts', appId, 'public', 'data', 'subscribers'), {
                    email: document.getElementById('subscriber-email').value,
                    timestamp: serverTimestamp()
                });
                msg.textContent = "SUCCESS. DATA CAPTURED.";
                msg.className = "mt-8 text-green-500 block font-bold tracking-widest text-xs uppercase";
                form.reset();
            } catch (err) {
                btn.disabled = false;
            }
        };

        const admin = document.getElementById('admin-panel');
        const checkHash = () => { if(window.location.hash === '#admin') admin.classList.remove('hidden'); };
        window.onhashchange = checkHash;
        checkHash();
        document.getElementById('close-admin').onclick = () => { admin.classList.add('hidden'); window.location.hash = ''; };

        document.getElementById('verify-admin').onclick = async () => {
            if(document.getElementById('admin-key').value === 'systems') {
                document.getElementById('admin-auth').classList.add('hidden');
                document.getElementById('admin-content').classList.remove('hidden');
                const snap = await getDocs(collection(db, 'artifacts', appId, 'public', 'data', 'subscribers'));
                const list = document.getElementById('admin-content');
                list.innerHTML = '';
                snap.forEach(doc => {
                    const d = doc.data();
                    list.innerHTML += `<div class="p-8 border border-white/5 rounded-2xl flex justify-between items-center font-mono">
                        <span class="text-white text-lg">${d.email}</span>
                        <span class="text-gray-600 text-xs">${d.timestamp ? new Date(d.timestamp.seconds*1000).toLocaleDateString() : ''}</span>
                    </div>`;
                });
            }
        };
    </script>
